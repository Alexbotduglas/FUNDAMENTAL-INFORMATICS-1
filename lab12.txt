int power(int x, int n)
	{
		int a = 1;
		while (n) 
		{
			if (n % 2) 
			{
				a *= x;
				n--;
			}

			else 
			{
				x *= x;
				n /= 2;
			}
		}
		
		return a;
		_getch();
	}
